exist graphics and child shape together

var stage = new Stage(canvas);
var sprite = new Sprite();
sprite.graphics.beginFill(0x0581d2);
sprite.graphics.drawCircle(0, 0, 40);
sprite.graphics.endFill();
sprite.x = 150;
sprite.y = 150;
stage.addChild(sprite);
var shape = new Shape();
shape.graphics.beginFill(0xffff00, 0.8);
shape.graphics.drawRegularStar(40, 0, 30);
shape.graphics.endFill();
sprite.addChild(shape);
var spriteDirection = 1;
stage.addEventListener('enterFrame', function (e) {
  if (sprite.x <= 75) {
    spriteDirection = 1;
  } else if (sprite.x >= 225) {
    spriteDirection = -1
  }
  sprite.x += spriteDirection * 1;

  shape.rotation++;
});

var stage = new Stage(canvas);
var sprite = new Sprite();
sprite.graphics.beginFill(0x00ff00);
sprite.graphics.drawRegularStar(0, 0, 80);
sprite.graphics.endFill();
var blue = new Shape();
blue.graphics.beginFill(0x0581d2, 0.5);
blue.graphics.drawCircle(0, 0, 50);
blue.graphics.endFill();
blue.x = blue.y = -20;
sprite.addChild(blue);
var red = new Shape();
red.graphics.beginFill(0xff0000, 0.5);
red.graphics.drawCircle(0, 0, 50);
red.graphics.endFill();
red.x = red.y = 20;
sprite.addChild(red);
sprite.x = 150;
sprite.y = 150;
stage.addChild(sprite);
var xDirection = 1;
var yDirection = -1;
stage.addEventListener('enterFrame', function (e) {
  if (sprite.scaleX >= 2) {
    xDirection = -1;
  } else if (sprite.scaleX <= 0.5) {
    xDirection = 1;
  }
  if (sprite.scaleY >= 1.5) {
    yDirection = -1;
  } else if (sprite.scaleY <= 0.5) {
    yDirection = 1;
  }
  sprite.scaleX += 0.01 * xDirection;
  sprite.scaleY += 0.01 * yDirection;
});
