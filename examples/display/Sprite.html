var stage = new Stage(canvas);
var sprite = new Sprite();
sprite.drawRegularStar(0, 0, 80);
sprite.fill(0x00ff00);
var blue = new Shape();
blue.drawCircle(0, 0, 50);
blue.fill(0x0581d2, 0.5);
blue.x = blue.y = -20;
sprite.addChild(blue);
var red = new Shape();
red.drawCircle(0, 0, 50);
red.fill(0xff0000, 0.5);
red.x = red.y = 20;
sprite.addChild(red);
sprite.x = 75;
sprite.y = 150;
stage.addChild(sprite);
var direction = 1;
stage.addEventListener('enterFrame', function (e) {
  if (sprite.x <= 75) {
    direction = 1;
  } else if (sprite.x >= 225) {
    direction = -1
  }
  sprite.x += direction * 1;
});

scale

var stage = new Stage(canvas);
var sprite = new Sprite();
sprite.drawRegularStar(0, 0, 80);
sprite.fill(0x00ff00);
var blue = new Shape();
blue.drawCircle(0, 0, 50);
blue.fill(0x0581d2, 0.5);
blue.x = blue.y = -20;
sprite.addChild(blue);
var red = new Shape();
red.drawCircle(0, 0, 50);
red.fill(0xff0000, 0.5);
red.x = red.y = 20;
sprite.addChild(red);
sprite.x = 150;
sprite.y = 150;
stage.addChild(sprite);
var xDirection = 1;
var yDirection = -1;
stage.addEventListener('enterFrame', function (e) {
  if (sprite.scaleX >= 2) {
    xDirection = -1;
  } else if (sprite.scaleX <= 0.5) {
    xDirection = 1;
  }
  if (sprite.scaleY >= 1.5) {
    yDirection = -1;
  } else if (sprite.scaleY <= 0.5) {
    yDirection = 1;
  }
  sprite.scaleX += 0.01 * xDirection;
  sprite.scaleY += 0.01 * yDirection;
});
