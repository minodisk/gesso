function onEnterFrame(e) {
  var matrix = new Matrix().createGradientBox(400, 200, angle, 0, 0);
  shape.clear();

  shape.drawRect(0, 0, 400, 200);
  shape.endFill();
  angle += Math.PI / 180;
}

public class Gradient extends Sprite {
  private var angle:Number;

  public function Gradient() {
    stage.scaleMode = StageScaleMode.NO_SCALE;
    stage.align = StageAlign.TOP_LEFT;
    angle = 0;
    addEventListener(Event.ENTER_FRAME, onEnterFrame);
  }

  private function onEnterFrame(e:Event):void {
    var matrix:Matrix = new Matrix();
    matrix.createGradientBox(400, 200, angle, 0, 0);
    graphics.clear();
    graphics.beginGradientFill(GradientType.LINEAR, [0xff0000, 0x0000ff], [1, 1], [0x00, 0xff], matrix);
    graphics.drawRect(0, 0, 400, 200);
    graphics.endFill();
    angle += Math.PI / 180;
  }
}
