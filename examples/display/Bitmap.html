display/Bitmap

draw

Assign file path.

var stage = new Stage(canvas);
var bitmap = new Bitmap();
stage.addChild(bitmap);
bitmap.draw('resources/lena.png');

Assign Image object that loads content incompletely.

var stage = new Stage(canvas);
var bitmap = new Bitmap();
stage.addChild(bitmap);
var img = new Image();
img.src = 'resources/opaque_dst.png';
bitmap.draw(img);

Assign Image object that loads content completely.

var stage = new Stage(canvas);
var bitmap = new Bitmap();
stage.addChild(bitmap);
var img = new Image();
img.addEventListener('load', function () {
  bitmap.draw(img);
}, false);
img.src = 'resources/lena.png';
