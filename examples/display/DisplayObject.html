translate

var stage = new Stage(canvas);
var shape = new Shape();
shape.drawRegularStar(0, 0, 50);
shape.fill(0x0581d2);
shape.x = shape.y = 75;
stage.addChild(shape);
var direction = 1;
stage.addEventListener('enterFrame', function (e) {
  if (shape.x <= 75) {
    direction = 1;
  } else if (shape.x >= 225) {
    direction = -1
  }
  shape.x += direction * 1;
});

scale

var stage = new Stage(canvas);
var shape = new Shape();
shape.drawRegularStar(0, 0, 50);
shape.fill(0x0581d2);
shape.x = 150;
shape.y = 75;
stage.addChild(shape);
var xDirection = 1;
var yDirection = -1;
stage.addEventListener('enterFrame', function (e) {
  if (shape.scaleX >= 1.5) {
    xDirection = -1;
  } else if (shape.scaleX <= 0.5) {
    xDirection = 1;
  }
  if (shape.scaleY >= 1.5) {
    yDirection = -1;
  } else if (shape.scaleY <= 0.5) {
    yDirection = 1;
  }
  shape.scaleX += 0.01 * xDirection;
  shape.scaleY += 0.01 * yDirection;
});
