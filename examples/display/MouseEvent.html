function onMouseEvent(e) {
  texts.push(e.eventPhase + ' ' + e.type +
    ' at (' + Math.round(e.localX) + ', ' + Math.round(e.localY) + ')');
  var container = (e.target.name === 'outer') ? dummyOuter :
                  (e.target.name === 'inner') ? dummyInner :
                  null;
  if (container) {
    container.pointer.x = e.localX;
    container.pointer.y = e.localY;
  }
  texts = texts.slice(-8);
  textField.text = texts.join('\n');
}
