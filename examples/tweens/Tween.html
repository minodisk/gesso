  'easeOutCubic',

  'easeOutInBounce',
  'easeInElastic',
  'easeOutElastic',
  'easeInOutElastic',
  'easeOutInElastic'
];
var points = [
  { x: 20, y: 20 },
  { x: 280, y: 20 },
  { x: 280, y: 280 },
  { x: 20, y: 280 }
];
var textField = new TextField();
var textFormat = new TextFormat();
textFormat.align = TextFormatAlign.CENTER;
textFormat.baseline = TextFormatBaseline.MIDDLE;
textField.textFormat = textFormat;
textField.x = textField.y = 150;
stage.addChild(textField);
var shape = new Shape();
shape.graphics.beginFill(0x0581d2);
shape.graphics.drawCircle(0, 0, 5);
shape.graphics.endFill();
shape.x = shape.y = 20;
stage.addChild(shape);
var tweens = [];
for (var i = 0, name; i < easingList.length; i++) {
  name = easingList[i];
  tweens.push(
    Tween.func(function (actor) {
      textField.text = actor.params[0];
    }, name),
    Tween.wait(500),
    Tween.to(shape, points[(i + 1) % 4], 1000, Easing[name]),
    Tween.wait(500)
  );
}
var tween = Tween.serial.apply(Tween, tweens);
tween.onComplete = function () {
  textField.text = 'Complete!';
};
tween.play();
