var stage = new Stage(canvas);
var shape;
var easingList = [
  Easing.linear,
  Easing.easeInQuad,
  Easing.easeOutQuad,
  Easing.easeInOutQuad,
  Easing.easeOutInQuad,
  Easing.easeInCubic,
  Easing.easeOutCubic,
  Easing.easeInOutCubic,
  Easing.easeOutInCubic
];
for (var i = 0; i < easingList.length; i++) {
  shape = new Shape();
  shape.drawCircle(0, 0, 5);
  shape.fill(0x0581d2);
  shape.x = 20;
  shape.y = 20 * (i + 1);
  stage.addChild(shape);
  Tween.to(shape, { x: 280 }, 1, easingList[i]).play();
}
