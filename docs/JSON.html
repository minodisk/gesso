    when 'number'

__walk = (holder, key, reviver) ->
  value = holder[key]
  if value and typeof value is 'object'
    for k of value
      v = __walk(value, k, reviver)
      if v isnt undefined
        value[k] = v
      else
        delete value[k]
  reviver.call(holder, key, value)
